@app.route('/zoning', methods=['POST'])
def zoning():
    lot_size = float(request.form['lot_size'])
    frontage = float(request.form['frontage'])
    city = request.form['city']

    results = []

    if lot_size >= 3000 and frontage >= 33:
        results.append("✅ Eligible for 4-plex under Bill 44")
        if lot_size >= 4000 and frontage >= 40:
            results.append("✅ May qualify for 6-plex with rental tenure or near transit")
    else:
        results.append("❌ Lot does not meet minimum size/frontage for multiplex")

    if city.lower() not in ['maple ridge', 'burnaby', 'coquitlam', 'langley', 'mission']:
        results.append("⚠️ Zoning rules may differ in your city. Please confirm with local planning office.")

    return render_template("zoning_result.html", results=results)
